---
title: Sign-In with Password
description: This guide explains how to implement a sign-in flow using email and password.
---

```tsx title="components/signin-form-field.tsx"
import { useSignIn, useTernSecure } from "@tern-secure/nextjs";

const { signIn } = useSignIn();
const { createActiveSession } = useTernSecure();

  const signInPasswordField = async () => {
    const res = await signIn?.withEmailAndPassword({ email, password });
    if (res?.status === "error") {
      setFormError({
        status: "error",
        message: res.message,
        error: res.error,
      });
    }
    if (res?.status === "success") {
      createActiveSession({ session: res.user, redirectUrl: afterSignInUrl });
    }
  };
```
